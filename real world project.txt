import pandas as pd
import yfinance as yf
import matplotlib.pyplot as plt

# 1. Fetch Real-World Data (Google stock for the last year)
ticker = 'GOOGL'
df = yf.download(ticker, period='1y')

# 2. Data Cleaning & Feature Engineering
# Calculate 20-day Moving Average to smooth trends
df['MA20'] = df['Close'].rolling(window=20).mean()

# 3. Basic Analysis
print(f"--- {ticker} Statistics ---")
print(df['Close'].describe())

# 4. Visualization: Price Trend & Moving Average
plt.figure(figsize=(12, 6))
plt.plot(df['Close'], label='Close Price', color='blue', alpha=0.5)
plt.plot(df['MA20'], label='20-Day Trend', color='red', linewidth=2)
plt.title(f'{ticker} Stock Price Performance (Past Year)')
plt.xlabel('Date')
plt.ylabel('Price (USD)')
plt.legend()
plt.grid(True)
plt.show()

# 5. Conclusion
latest_price = df['Close'].iloc[-1].values[0]
print(f"\nConclusion: Current {ticker} price is ${latest_price:.2f}.")
